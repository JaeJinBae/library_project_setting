-- 반납된 경우
-- 대여테이블 정보 삭제 후 반납테이블 insert
-- 연체확인후 user update

use library;$$

create trigger rentalBook_delete_tri
before delete on rentalBook
for each row 
begin
   
   declare curdays int;
   set curdays = datediff(old.rentalDay,curdate());
   
   if(curdays <0) then
      update users
      set delay_day = -curdays
      where user_Id = old.user_Id;
   end if;
   insert into history
   values(old.book_Code,old.user_id,old.rentalDay,curdate());
end $$
